# Sheet

Extends the Dialog component to display content that complements the main screen.

<FigmaEmbed url="https://www.figma.com/file/YOUR_FILE_ID/Sheet" />

## Usage

```tsx
import {
  Sheet,
  SheetTrigger,
  SheetContent,
  SheetHeader,
  SheetTitle,
  SheetDescription,
} from '@emmettorama/substance';
```

```tsx
<Sheet>
  <SheetTrigger asChild>
    <Button variant="outline">Open Sheet</Button>
  </SheetTrigger>
  <SheetContent>
    <SheetHeader>
      <SheetTitle>Edit profile</SheetTitle>
      <SheetDescription>
        Make changes to your profile here.
      </SheetDescription>
    </SheetHeader>
    {/* Your form content here */}
  </SheetContent>
</Sheet>
```

## Side

```tsx
<SheetContent side="left">Left sidebar</SheetContent>
<SheetContent side="right">Right sidebar</SheetContent>
<SheetContent side="top">Top panel</SheetContent>
<SheetContent side="bottom">Bottom panel</SheetContent>
```

## API

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | — | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | — | Change handler |
| `side` | `"top" \| "right" \| "bottom" \| "left"` | `"right"` | Slide-in side |
